# Chapter 5 Practice Set - Dictionary & Sets

# Q1: Create a dictionary of Hindi words with meanings
d = {
    "pankha": "fan",
    "dabba": "box",
    "kursi": "chair"
}
print(d)
# Output: {'pankha': 'fan', 'dabba': 'box', 'kursi': 'chair'}

# Q2: Input 8 numbers and display unique values
nums = {1, 2, 2, 3, 4, 4, 5, 6}
print(nums)
# Output: {1, 2, 3, 4, 5, 6}

# Q3: Check type of set
s = {18, "18"}
print(s)
# Output: {18, '18'}

# Q4: Create empty dictionary and add 4 friends' names and languages
friends = {}
friends["A"] = "Python"
friends["B"] = "Java"
friends["C"] = "C++"
friends["D"] = "JavaScript"
print(friends)

# Q5: Same keys but different values
d = {"name": "Kashish", "Name": "Shaikh"}
print(d)
# Output: {'name': 'Kashish', 'Name': 'Shaikh'}
